# 配置 crictl 工具
cat > /etc/crictl.yaml << 'EOF'
runtime-endpoint: unix:///run/containerd/containerd.sock
image-endpoint: unix:///run/containerd/containerd.sock
timeout: 30
debug: false
pull-image-on-create: false
EOF

# 锁定版本禁止apt自动更新
sudo apt-mark hold kubelet kubeadm kubectl;

# 校验版本
kubeadm version;
kubectl version --client;
kubelet --version;

# 预拉取 kube-vip 镜像
sudo ctr images pull ghcr.m.daocloud.io/kube-vip/kube-vip:v1.2.1;
# 镜像重命名避免海外地址拉取
sudo ctr images tag ghcr.m.daocloud.io/kube-vip/kube-vip:v1.2.1 ghcr.io/kube-vip/kube-vip:v1.2.1;

# 生成的yaml写入静态Pod目录
sudo ctr run --rm --net-hostghcr.m.daocloud.io/kube-vip/kube-vip:v1.2.1 vip-generator \
/kube-vip manifest pod \
--interface ens33 \
--vip 192.168.120.100 \
-- vip_subnet 24   \   #待定
--controlplane \
--services \
--arp | sudo tee /etc/kubernetes/manifests/kube-vip.yaml;

# 校验yaml生成成功
ls /etc/kubernetes/manifests/;
cat /etc/kubernetes/manifests/kube-vip.yaml | grep $VIP;


#预拉取集群所有组件镜像
sudo kubeadm config images pull --config kubeadm-init-config.yaml;
# 校验镜像全部拉取完成
sudo crictl images | grep registry.aliyuncs.com/google_containers;


# 执行集群初始化
sudo kubeadm init --config kubeadm-init.yaml --upload-certs | tee kubeadm-init-output.log0;

# init-token过期重新生成
## 先上传证书，拿到 certificate-key
kubeadm init phase upload-certs --upload-certs

## 生成永久有效 worker 基础 join 命令
kubeadm token create --ttl 0 --print-join-command

###示例（master）
kubeadm join 192.168.120.100:6443 \
--token iz5fxg.1g3sod1v1hluabxw \
--discovery-token-ca-cert-hash sha256:305b861fa5e83ac3ca66e1566bae643dbeedd639f630035585d92b187ad2e476 \
--control-plane --certificate-key 你的certificate-key

# 卸载节点
kubeadm reset -f
rm -rf /etc/kubernetes /var/lib/kubelet /var/lib/etcd
rm -rf /etc/systemd/system/kubelet.service.d /etc/cni/net.d/*
